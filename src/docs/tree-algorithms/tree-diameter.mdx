## Code

```cpp lineNumbers=true
#include<bits/stdc++.h>
using namespace std;
 
const int maxN = 2e5+5;
int n, u, v, d[maxN];
vector<int> adj[maxN];
int ans = 0;
 
void dfs(int s, int p) {
    for(auto u: adj[s]) {
        if(u == p) continue;
        dfs(u, s);
        ans = max(ans, d[s] + d[u] + 1);
        d[s] = max(d[s], d[u]+1);
    }
}
 
void solve() {
    cin >> n;
    for(int i=2; i<=n; i++) {
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
 
    dfs(1, 0);
    cout << ans << "\n";
}
 
int main() {
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    solve();
    return 0;
}
```
